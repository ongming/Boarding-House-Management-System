-- 1. Xóa database cũ nếu đã tồn tại để làm sạch dữ liệu
DROP DATABASE IF EXISTS BoardingHouse_Pro;

-- 2. Tạo mới Database
CREATE DATABASE BoardingHouse_Pro;
USE BoardingHouse_Pro;

-- 3. Bảng TAI_KHOAN (Hệ thống dùng để xác thực Đăng nhập & Phân quyền)
CREATE TABLE TAI_KHOAN (
    id INT AUTO_INCREMENT PRIMARY KEY,
    ten_dang_nhap VARCHAR(50) NOT NULL UNIQUE,
    mat_khau VARCHAR(255) NOT NULL,
    ho_ten VARCHAR(100),
    loai_tai_khoan ENUM('ADMIN', 'NHAN_VIEN') NOT NULL
);

-- 4. Bảng NHAN_VIEN (Thông tin bổ sung cho nhân viên)
CREATE TABLE NHAN_VIEN (
    id INT AUTO_INCREMENT PRIMARY KEY,
    tai_khoan_id INT UNIQUE,
    ca_lam_viec TEXT, -- Lưu lịch làm việc từ AI xếp ca
    FOREIGN KEY (tai_khoan_id) REFERENCES TAI_KHOAN(id) ON DELETE CASCADE
);

-- 5. Bảng PHONG (Quản lý Tầng, Loại máy lạnh và Nội thất)
CREATE TABLE PHONG (
    id INT AUTO_INCREMENT PRIMARY KEY,
    so_phong VARCHAR(10) NOT NULL UNIQUE,
    tang INT CHECK (tang IN (1, 2)),
    co_may_lanh BOOLEAN DEFAULT FALSE,
    gia_thue_co_ban DECIMAL(15, 2) NOT NULL,
    danh_sach_noi_that TEXT, -- Ghi chú: Giường, Tủ, Máy lạnh...
    trang_thai ENUM('TRONG', 'DA_THUE', 'BAO_TRI') DEFAULT 'TRONG'
);

-- 6. Bảng KHACH_THUE (Thông tin cá nhân khách đại diện)
CREATE TABLE KHACH_THUE (
    id INT AUTO_INCREMENT PRIMARY KEY,
    ho_ten VARCHAR(100) NOT NULL,
    so_cccd VARCHAR(20) UNIQUE NOT NULL,
    so_dien_thoai VARCHAR(15)
);

-- 7. Bảng HOP_DONG (Lưu thông tin thuê, ảnh hợp đồng và số người ở)
CREATE TABLE HOP_DONG (
    id INT AUTO_INCREMENT PRIMARY KEY,
    khach_thue_id INT,
    phong_id INT,
    ngay_bat_dau DATE NOT NULL,
    ngay_ket_thuc DATE,
    tien_coc DECIMAL(15, 2),
    anh_hop_dong_url VARCHAR(255), -- Lưu đường dẫn file ảnh CCCD/Hợp đồng
    so_luong_nguoi_o INT DEFAULT 1,
    trang_thai ENUM('HIEU_LUC', 'KET_THUC') DEFAULT 'HIEU_LUC',
    FOREIGN KEY (khach_thue_id) REFERENCES KHACH_THUE(id),
    FOREIGN KEY (phong_id) REFERENCES PHONG(id)
);

-- 8. Bảng XE (Để quản lý biển số và tính phí xe hàng tháng)
CREATE TABLE XE (
    id INT AUTO_INCREMENT PRIMARY KEY,
    hop_dong_id INT,
    loai_xe VARCHAR(50), -- 'Xe máy', 'Ô tô'
    bien_so VARCHAR(20),
    FOREIGN KEY (hop_dong_id) REFERENCES HOP_DONG(id) ON DELETE CASCADE
);

-- 9. Bảng CHI_SO_DIEN_NUOC (Lưu lịch sử chốt số hàng tháng)
CREATE TABLE CHI_SO_DIEN_NUOC (
    id INT AUTO_INCREMENT PRIMARY KEY,
    phong_id INT,
    thang INT NOT NULL,
    nam INT NOT NULL,
    so_dien_cu INT NOT NULL,
    so_dien_moi INT NOT NULL,
    so_nuoc_cu INT NOT NULL,
    so_nuoc_moi INT NOT NULL,
    ngay_ghi DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (phong_id) REFERENCES PHONG(id)
);

-- 10. Bảng HOA_DON (Lưu chi tiết từng khoản phí, hỗ trợ Bồi thường & Ghi chú)
CREATE TABLE HOA_DON (
    id INT AUTO_INCREMENT PRIMARY KEY,
    hop_dong_id INT,
    nhan_vien_id INT,
    thang INT NOT NULL,
    nam INT NOT NULL,
    tien_phong DECIMAL(15, 2) DEFAULT 0,
    tien_dien DECIMAL(15, 2) DEFAULT 0,
    tien_nuoc DECIMAL(15, 2) DEFAULT 0,
    tien_xe DECIMAL(15, 2) DEFAULT 0,
    tien_rac DECIMAL(15, 2) DEFAULT 0,
    tien_boi_thuong DECIMAL(15, 2) DEFAULT 0, -- Cột bổ sung cho Checkout
    tong_tien DECIMAL(15, 2) NOT NULL,
    da_thanh_toan BOOLEAN DEFAULT FALSE,
    ghi_chu TEXT, -- Ghi nội dung hư hại thiết bị
    loai_hoa_don ENUM('DIN_KY', 'BOI_THUONG', 'QUYET_TOAN') DEFAULT 'DIN_KY',
    ngay_lap DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (hop_dong_id) REFERENCES HOP_DONG(id),
    FOREIGN KEY (nhan_vien_id) REFERENCES NHAN_VIEN(id)
);

-- 11. Bảng PHAN_HOI (Ghi nhận sự cố từ NV và hướng giải quyết từ Admin)
CREATE TABLE PHAN_HOI (
    id INT AUTO_INCREMENT PRIMARY KEY,
    phong_id INT,
    nhan_vien_id INT,
    noi_dung TEXT,
    trang_thai ENUM('CHO_XU_LY', 'DANG_SUA', 'HOAN_THANH') DEFAULT 'CHO_XU_LY',
    ngay_gui DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (phong_id) REFERENCES PHONG(id),
    FOREIGN KEY (nhan_vien_id) REFERENCES NHAN_VIEN(id)
);

-- 12. Bảng CAU_HINH_GIA (Admin quản lý đơn giá tại đây)
CREATE TABLE CAU_HINH_GIA (
    id INT AUTO_INCREMENT PRIMARY KEY,
    ten_loai VARCHAR(50) UNIQUE, -- 'DIEN', 'NUOC', 'RAC', 'XE_MAY'
    don_gia DECIMAL(15, 2) NOT NULL
);
